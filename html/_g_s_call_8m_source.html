136 - (void)callStateDidChange:(NSNotification *)notif {
137  pjsua_call_id callId = GSNotifGetInt(notif, GSSIPCallIdKey);
138  pjsua_acc_id accountId = GSNotifGetInt(notif, GSSIPAccountIdKey);
139  if (callId != _callId || accountId != _account.accountId)
140  return;
141 
142  pjsua_call_info callInfo;
143  pjsua_call_get_info(_callId, &callInfo);
144 
145  GSCallStatus callStatus;
146  switch (callInfo.state) {
147  case PJSIP_INV_STATE_NULL: {
148  callStatus = GSCallStatusReady;
149  } break;
150 
151  case PJSIP_INV_STATE_CALLING:
152  case PJSIP_INV_STATE_INCOMING: {
153  callStatus = GSCallStatusCalling;
154  } break;
155 
156  case PJSIP_INV_STATE_EARLY:
157  case PJSIP_INV_STATE_CONNECTING: {
158  callStatus = GSCallStatusConnecting;
159  } break;
160 
161  case PJSIP_INV_STATE_CONFIRMED: {
162  callStatus = GSCallStatusConnected;
163  } break;
164 
165  case PJSIP_INV_STATE_DISCONNECTED: {
166  callStatus = GSCallStatusDisconnected;
167  } break;
168  }
169 
170  __block id self_ = self;
171  dispatch_async(dispatch_get_main_queue(), ^{ [self_ setStatus:callStatus]; });
172 }

174 - (void)callMediaStateDidChange:(NSNotification *)notif {
175  pjsua_call_id callId = GSNotifGetInt(notif, GSSIPCallIdKey);
176  if (callId != _callId)
177  return;
178 
179  pjsua_call_info callInfo;
180  GSReturnIfFails(pjsua_call_get_info(_callId, &callInfo));
181 
182  if (callInfo.media_status == PJSUA_CALL_MEDIA_ACTIVE) {
183  pjsua_conf_port_id callPort = pjsua_call_get_conf_port(_callId);
184  GSReturnIfFails(pjsua_conf_connect(callPort, 0));
185  GSReturnIfFails(pjsua_conf_connect(0, callPort));
186 
187  [self adjustVolume:_volume mic:_micVolume];
188  }
189 }

192 - (BOOL)adjustVolume:(float)volume mic:(float)micVolume {
193  GSAssert(0.0 <= volume && volume <= 1.0, @"Volume value must be between 0.0 and 1.0");
194  GSAssert(0.0 <= micVolume && micVolume <= 1.0, @"Mic Volume must be between 0.0 and 1.0");
195 
196  _volume = volume;
197  _micVolume = micVolume;
198  if (_callId == PJSUA_INVALID_ID)
199  return YES;
200 
201  pjsua_call_info callInfo;
202  pjsua_call_get_info(_callId, &callInfo);
203  if (callInfo.media_status == PJSUA_CALL_MEDIA_ACTIVE) {
204 
205  // scale volume so 1.0 is 2x louder
206  volume *= _volumeScale;
207  micVolume *= _volumeScale;
208  pjsua_conf_port_id callPort = pjsua_call_get_conf_port(_callId);
209  GSReturnNoIfFails(pjsua_conf_adjust_rx_level(callPort, volume));
210  GSReturnNoIfFails(pjsua_conf_adjust_tx_level(callPort, micVolume));
211  }
212 
213  // send volume change notification
214  NSDictionary *info = nil;
215  info = [NSDictionary dictionaryWithObjectsAndKeys:
216  [NSNumber numberWithFloat:volume], GSVolumeKey,
217  [NSNumber numberWithFloat:micVolume], GSMicVolumeKey, nil];
218 
219  NSNotificationCenter *center = [NSNotificationCenter defaultCenter];
220  [center postNotificationName:GSVolumeDidChangeNotification
221  object:self
222  userInfo:info];
223 
224  return YES;
225 }
